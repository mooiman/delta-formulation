\subsection{Essential/natural boundary, various terms}

For the essential and natural boundary condition the boundary condition is precribed via a line integral.

%--------------------------------------------------------------------------------
Consider the line integral needed along the boundary, for a left/west boundary we have $\xi = \it{Constant}$.
\begin{align}
    \int_\Gamma \vec{f}(x,y) \, dS
\end{align}
where $\Gamma$ is the curve describing the boundary.
For curvilinear coordinates the distance formulation reads (using the metric tensor, \citet{Aris1962}):
\begin{align}
    (ds)^2 = g_{\xi\xi}\,(d\xi)^2 + 2 g_{\xi\eta}\,d\xi d\eta + g_{\eta \eta}\, (d\eta)^2,
\end{align}
with
\begin{align}
g_{\xi\xi} &= x_\xi x_\xi + y_\xi y_\xi \\
g_{\xi\eta} & = x_\xi x_\eta  + y_\xi y_\eta = g_{\eta\xi} \\
g_{\eta\eta} &= x_\eta x_\eta  + y_\eta y_\eta
\end{align}
The line intgral in curvilinear coordinates reads:
\begin{align}
    \int^{\paramt_1}_{\paramt_0}  \vec{f}(\xi,\eta) ds \, d\paramt  = \int^{\paramt_1}_{\paramt_0} \vec{f}(\xi,\eta) \, \sqrt{ g_{\xi\xi} \left(\diff{\xi}{\paramt}\right)^2
        + 2g_{\xi\eta} \diff{\xi}{\paramt} \diff{\eta}{\paramt}
        + g_{\eta\eta} \left(\diff{\eta}{\paramt}\right)^2 } \, d\paramt
\end{align}
with the parametrization of the curve is for $\paramt \in [0,1]$.

\paragraph*{Essential and natural boundary location}
The essential boundary is located at a location in the control volume $0\leq\xi\leq 1$ (see \autoref{fig:structured_grid_along_straight_boundary}, where $\xi \approx \frac{1}{3}$) and the natural boundary is located at the middle of an element (edge of a sub-control volume).

In the remainder of this section we will focus on the left/west boundary of a control volume (i.e.\ natural boundary, same derivation holds for the essential boundary condition).
Similar approaches are suitable for the boundary at the other three sides of the structured grid and also for the various terms of the equations.

The natural boundary is on the edge of a control volume, $\xi = \half$.
We consider a part on the left/west boundary from $[j - \half, j]$ and with the curve parametrization for $j = 1$ and $\paramt \in [0,1]$:
%
\begin{alignat}{3}
    \begin{aligned}
    \xi(\paramt) &= \half      &\quad \Rightarrow \diff{\xi(\paramt)}{\paramt} = 0 \\
    \eta(\paramt) &= \half + \half \paramt &\quad \Rightarrow \diff{\eta(\paramt)}{\paramt} = \half
    \end{aligned}
    \label{eq:parametrization}
\end{alignat}
Using this parametrization for a general function $\vec{f}$ yields
\begin{align}
    &\int_\Gamma  \vec{f}(x,y)\, dS
%    = \int^1_0 \vec{f}(\xi,\eta) ds \, d\paramt
    = \sqrt{g_{\eta\eta} \left(\half\right)^2} \,\vec{f}(\xi,\eta)
    = \half \sqrt{g_{\eta\eta}} \vec{f}(\xi,\eta)
\end{align}
%
This parametrization will be used in the remainder of this section.

%------------------------------------------------------------------------------
\subsubsection{Time derivative}
The time derivative for $h$ reads
\begin{align}
    \int_\Gamma \pdiff{h}{t} \, dS = \half \sqrt{g_{\eta\eta}} \pdiff{h}{t}
\end{align}
and is similar for the discharges $q$ and $r$.

%------------------------------------------------------------------------------
\subsubsection{Mass flux}
The mass flux in the continuity equation reads:
\begin{align}
    \nabla \dotp \vec{q} = \pdiff{q}{x} + \pdiff{r}{x}
\end{align}
Taking the line integral at the boundary yields:
\begin{align}
    \int_\Gamma \nabla \dotp \vec{q}\, ds & = \half \sqrt{g_{\eta\eta}} \left( \pdiff{q}{x} + \pdiff{r}{y} \right) =
\\
%    & = \frac{1}{J} \Bigg(
%    y_{\eta} \pdiff{q}{\xi}- y_{\xi}\pdiff{q}{\eta} + {\color{blue}y_{\eta\xi}q - y_{\xi\eta}q}
%    \\
%    & \qquad - x_{\eta}\pdiff{r}{\xi} + x_{\xi}\pdiff{r}{\eta} +{\color{blue}x_{\eta\xi}r - x_{\xi\eta}r} \Bigg)
%    \\
     & = \half \sqrt{g_{\eta\eta}} \left[ \frac{1}{J}\Bigg( \pdiff{}{\xi} \left( y_{\eta} q - x_{\eta} r \right) + \pdiff{}{\eta} \left( -y_{\xi} q + x_{\xi} r \right)\Bigg) \right],
\end{align}
for the term between square brackets see \autoref{eq:full_curvilinear_mass_flux}.
%------------------------------------------------------------------------------
\subsubsection{Convection}
%------------------------------------------------------------------------------
\paragraph*{$q$-momentum}
The convection term in the $q$-momentum equation can be transformed as:
\begin{align}
     \int_\Gamma \textit{convection}\, dS &= \half \sqrt{g_{\eta\eta}} \left[\frac{1}{J} \pdiff{}{\xi}\left(y_{\eta} \frac{q^2}{h} - x_{\eta} \frac{qr}{h} \right) + \frac{1}{J} \pdiff{}{\eta}\left(-y_{\xi} \frac{q^2}{h} + x_{\xi} \frac{qr}{h} \right) \right].
\end{align}
for the term between square brackets see \autoref{eq:xi_convection_term}.
%------------------------------------------------------------------------------
\paragraph*{$q$-momentum}
The convection term in the $r$-momentum equation can be transformed as:
\begin{align}
     \int_\Gamma \textit{convection} \, dS &= \half \sqrt{g_{\eta\eta}} \left[\frac{1}{J} \pdiff{}{\xi}\left(y_{\eta} \frac{rq}{h} - x_{\eta} \frac{r^2}{h} \right) + \frac{1}{J} \pdiff{}{\eta}\left(-y_{\xi} \frac{rq}{h} + x_{\xi} \frac{r^2}{h} \right) \right].
\end{align}
for the term between square brackets see \autoref{eq:eta_convection_term}.
%------------------------------------------------------------------------------
\subsubsection{Bed shear stress}
%------------------------------------------------------------------------------
\paragraph*{$q$-momentum}
The bed shear stress term in the $q$-momentum equation can be transformed as:
\begin{align}
    \int_\Gamma \textit{bed\_shear\_stress} \, dS &= \half \sqrt{g_{\eta\eta}} \left[ c_f \left( \frac{q\abs{\widetilde{\vec{q}}}}{h^2} \right) \right].
\end{align}
for the term between square brackets see \autoref{eq:xi_bed_shear_stress_term} divided by $J$.
%------------------------------------------------------------------------------
\paragraph*{$r$-momentum}
The bed shear stress term in the $r$-momentum equation can be transformed as:
\begin{align}
    \int_\Gamma \textit{bed\_shear\_stress} \, dS &= \half \sqrt{g_{\eta\eta}} \left[ c_f \left( \frac{r\abs{\widetilde{\vec{q}}}}{h^2} \right) \right].
\end{align}
for the term between square brackets see \autoref{eq:eta_bed_shear_stress_term} divided by $J$.

\subsubsection{Viscosity}
%------------------------------------------------------------------------------
\paragraph*{$q$-momentum}
\vspace{-1\baselineskip}
\begin{align}
    \begin{aligned}
        \int_\Gamma \textit{viscosity} \, dS & = \half \sqrt{g_{\eta\eta}} \Bigg[
        \\
        - \frac{1}{J} \pdiff{}{\xi}\Bigg[ & \frac{\nu_{11} h}{J} \bigg(
        2 y_{\eta}  y_{\eta} \pdiff{(q/h)}{\xi}
        - 2 y_{\eta} y_{\xi} \pdiff{(q/h)}{\eta} \bigg)
        +
        \\
        + &
        \frac{\nu_{12} h}{J} \bigg(
        - x_{\eta}y_{\eta} \pdiff{(r/h)}{\xi}
        + x_{\eta}y_{\xi} \pdiff{(r/h)}{\eta}
        + x_{\eta}x_{\eta}\pdiff{(q/h)}{\xi}
        - x_{\eta}x_{\xi}\pdiff{(q/h)}{\eta} \bigg) \Bigg] +
        \\
        + \frac{1}{J} \pdiff{}{\eta}\Bigg[& \frac{\nu_{11} h}{J} \bigg(
        2 y_{\xi}y_{\eta} \pdiff{(q/h)}{\xi}
        - 2 y_{\xi}y_{\xi} \pdiff{(q/h)}{\eta} \bigg) +
        \\
        + &
        \frac{\nu_{12} h}{J} \bigg(
        - x_{\xi}y_{\eta} \pdiff{(r/h)}{\xi}
        + x_{\xi}y_{\xi} \pdiff{(r/h)}{\eta}
        + x_{\xi} x_{\eta} \pdiff{(q/h)}{\xi}
        - x_{\xi}x_{\xi} \pdiff{(q/h)}{\eta} \bigg) \Bigg]
        \Bigg]
    \end{aligned}
\end{align}
for the term between the outer square brackets see \autoref{eq:xi_viscous_term}.
